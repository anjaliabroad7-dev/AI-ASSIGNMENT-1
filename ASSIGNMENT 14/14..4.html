Here’s a simple Flask application that serves the login form (from Task #3) and prints the username on successful login. The app uses server‑side validation to ensure both fields are non‑empty. ✅ Flask App Code from flask import Flask, render_template_string, request app = Flask(__name__) # HTML template with form login_form_html = """ Login Form

Login

{% if username_error %}
{{ username_error }}
{% endif %} {% if password_error %}
{{ password_error }}
{% endif %}
{% if success %}

Login successful! Welcome, {{ username }}.

{% endif %}
""" @app.route("/", methods=["GET", "POST"]) def login(): username_error = "" password_error = "" success = False username = "" if request.method == "POST": username = request.form.get("username", "").strip() password = request.form.get("password", "").strip() if not username: username_error = "Username cannot be empty." if not password: password_error = "Password cannot be empty." if username and password: success = True return render_template_string( login_form_html, username=username, username_error=username_error, password_error=password_error, success=success ) if __name__ == "__main__": app.run(debug=True)
